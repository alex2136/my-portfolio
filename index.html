02 OOP Example — PlayerData
PlayerData.lua Luau 
-- PlayerData class
-- handles per-player data with autosave and safe defaults

local DataStoreService = game:GetService("DataStoreService")
local Players          = game:GetService("Players")

local Store = DataStoreService:GetDataStore("PlayerData_v1")

local PlayerData = {}
PlayerData.__index = PlayerData

local DEFAULTS = {
    Cash    = 0,
    Level   = 1,
    XP      = 0,
    Playtime = 0,
}

function PlayerData.new(player: Player)
    local self = setmetatable({}, PlayerData)

    self.Player    = player
    self.UserId    = player.UserId
    self._data     = {}
    self._loaded   = false
    self._joinTime = os.time()

    self:_load()
    return self
end

function PlayerData:_load()
    local ok, result = pcall(function()
        return Store:GetAsync(self.UserId)
    end)

    if ok and result then
        -- merge saved data with defaults so new keys are always present
        for k, v in pairs(DEFAULTS) do
            self._data[k] = result[k] ~= nil and result[k] or v
        end
    else
        self._data = table.clone(DEFAULTS)
    end

    self._loaded = true
end

function PlayerData:Get(key: string)
    return self._data[key]
end

function PlayerData:Set(key: string, value: any)
    self._data[key] = value
end

function PlayerData:Increment(key: string, amount: number)
    self._data[key] = (self._data[key] or 0) + amount
end

function PlayerData:Save()
    if not self._loaded then return end

    self._data.Playtime = self._data.Playtime + (os.time() - self._joinTime)
    self._joinTime = os.time()

    local ok, err = pcall(function()
        Store:SetAsync(self.UserId, self._data)
    end)

    if not ok then
        warn("[PlayerData] save failed for", self.UserId, err)
    end
end

function PlayerData:Destroy()
    self:Save()
    self._data = nil
end

return PlayerData
